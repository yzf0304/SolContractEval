/**
 * @title Ownable
 * @dev A simple contract to define an owner and provide authorization control functions.
 */
contract Ownable {

    address public owner = msg.sender;

    modifier onlyOwner() {
        require(msg.sender == owner);
        _;
    }

    function transferOwnership(address newOwner) public onlyOwner {
        require(newOwner != address(0));
        owner = newOwner;
    }
}
